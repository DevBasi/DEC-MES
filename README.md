def encode_message(message):
    """–®–∏—Ñ—Ä—É–µ—Ç —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –ø–æ–º–æ—â—å—é Fernet (AES) –∏ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç –∫–∞–∫ —Å—Ç—Ä–æ–∫—É —Å –º–Ω–æ–∂–µ—Å—Ç–≤–æ–º —Å–∏–º–≤–æ–ª–æ–≤."""
    key = load_key()
    f = Fernet(key)
    message_bytes = message.encode('utf-8')
    encrypted = f.encrypt(message_bytes)
    # –ü—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ–º –∫–∞–∂–¥—ã–π –±–∞–π—Ç –∫–∞–∫ —Å–∏–º–≤–æ–ª –∏–∑ –Ω–∞–±–æ—Ä–∞ symbols
    encoded_str = ''.join(symbols[b] for b in encrypted)
    return encoded_str

def decode_message(encoded_message):
    """–î–µ—à–∏—Ñ—Ä—É–µ—Ç —Å–æ–æ–±—â–µ–Ω–∏–µ –∏–∑ —Å—Ç—Ä–æ–∫–∏ —Å –º–Ω–æ–∂–µ—Å—Ç–≤–æ–º —Å–∏–º–≤–æ–ª–æ–≤."""

        key = load_key()
        f = Fernet(key)
        # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Å—Ç—Ä–æ–∫—É –æ–±—Ä–∞—Ç–Ω–æ –≤ –±–∞–π—Ç—ã, –∏—Å–ø–æ–ª—å–∑—É—è –∏–Ω–¥–µ–∫—Å –≤ symbols
        encoded_bytes = bytes(symbols.index(c) for c in encoded_message)
        decrypted = f.decrypt(encoded_bytes)

    except Exception as e:
